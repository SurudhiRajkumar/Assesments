import functools
import math
n=int(input("Enter:"))
num=[]
deno=[]
comm=[]
for i in range(0,n):
    for j in range(0,2):
        a=int(input())
        comm.append(a)
print(comm)
for i in range(0,len(comm),2):
    num.append(comm[i])
for i in range(1,len(comm)+1,2):
    deno.append(comm[i])
print(num)
print(deno)
pdt_num= functools.reduce(lambda x,y :x*y , num)
pdt_deno= functools.reduce(lambda x,y :x*y , deno)
print(pdt_num)
print(pdt_deno)
div=pdt_num/pdt_deno
div1=div.as_integer_ratio()
for i in range(0,len(div1)):
    print(div1[i],end="")
    if i!= len(div1):
        print("/", end="")
    else:
        continue

import functools
n=int(input("Enter:"))
name=input("enter:")
l1=[]
for i in range(n):
    a=input()
    l=a.split(" ")
    l1.append(l)
print(l1)
user_dict={}
for user in l1:
    mark_list = user[1::]
    avg = (functools.reduce(lambda x,y: int(x)+int(y), mark_list))/len(mark_list)
    user_dict[user[0]] = avg

print(user_dict[name])
